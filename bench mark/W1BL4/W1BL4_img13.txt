བྲམ་ཟེའི་ཁྱེའུ་རྣམས་ཀྱི་ཀུན་དུ་སྤྱོད་པ་ནི་འདི་ཡིན་ཏེ། གང་གི་ཚེ་མི་ཀློག་པར་གྱུར་པ་དེའི་ཚེ་ན་ལན་འགའ་ནི་ཆུའི་ངོགས་སུ་ཉེ་རེག་བྱེད་དུ་འདོང་ངོ་། །ལན་འགའ་ནི་གྲོང་ཁྱེར་ལྟར་འདོང་ངོ་། །ལན་འགའ་ནི་ཡམ་སྲེག་ཤིང་ཐུར་འདོང་ངོ་། །ཇི་ཙམ་དུས་གཞན་ཞིག་ན་དེ་དག་མ་ཀློག་
པར་གྱུར་ནས་དེའི་ཚེ་དེ་དག་ཡམ་སྲེག་ཤིང་ཐུར་དོང་བ་དང་། དེ་དག་ལ་ལམ་གྱི་བར་དུ་བར་སྐབས་ཀྱི་གཏམ་བརྗོད་པ་འདི་ལྟ་བུ་འདི་དག་བྱུང་སྟེ། ཀྱེ་ཀུད་སའི་བུ་དག །བད་སའི་བུ་དག །ཤན་དི་ལའི་བུ་དག །བ་ར་དྭ་ཛའི་བུ་དག །ལྔ་པའི་བུ་དག །ཉེ་ལྔ་པའི་བུ་དག་ཇེ་ཁྱེད་ཅག་ལས་གང་
ནི་ཡུལ་ག་ལས་འོངས་པ་དྲི་བར་བྱའོ། །དེ་ན་བྲམ་ཟེའི་ཁྱེའུ་གཅིག་གིས་སྨྲས་པ། ཁོ་བོ་ནི་ཡུལ་ཤར་ནས་སོ། །གཞན་ཞིག་གིས་སྨྲས་པ། ཁོ་བོ་ནི་ལྷོ་ཕྱོགས་ནས་སོ། །ཡང་གཞན་ཞིག་གིས་སྨྲས་པ། ཁོ་བོ་ནི་ནུབ་ཕྱོགས་ནས་སོ། །ཡང་གཞན་ཞིག་གིས་སྨྲས་པ། ཁོ་བོ་ནི་
བྱང་ཕྱོགས་ནས་སོ། །བྲམ་ཟེའི་ཁྱེའུ་དེས་སྨྲས་པ། ཁོ་བོ་ནི་ཡུལ་དབུས་ནས་སོ། །དེ་དག་གིས་སྨྲས་པ། ཤེས་ལྡན་དག་བདག་ཅག་གིས་ཡུལ་ཐམས་ཅད་མཐོང་ཞིང་ཐོས་ན་ཡུལ་དབུས་ནི་མ་མཐོང་མ་ཐོས་སོ། །ཤར་ན་བློ་ཅན་གནས་པ་སྟེ། །ལྷོ་ཕྱོགས་ན་ནི་གཡོན་བྱེད་འདུག །
ནུབ་ཀྱི་ཡུལ་ན་ཕྲ་མ་ཅན། །བྱང་ཕྱོགས་ན་ནི་ཚིག་རྩུབ་བོ། །ཞེས་ཚིགས་སུ་བཅད་པ་ཡང་སྨྲས་སོ། །དེ་ནས་དྲིས་པ། བྲམ་ཟེའི་ཁྱེའུ་ཡུལ་དབུས་ཅི་འདྲ། ཤེས་ལྡན་དག་ཡུལ་དབུས་ནི་ཡུལ་རྣམས་ཀྱི་མཆོག་ཡིན་ཏེ། སཱ་ལུ་དང་། བུ་རམ་ཤིང་གི་ཕྲེང་བ་དང་། བ་ལང་
དང་། མ་ཧེ་ཕུན་སུམ་ཚོགས་པ། སློང་མོ་པ་བརྒྱས་ཁྱབ་པ། དབུས་པའི་སྐྱེ་བོས་གང་བ། མཐའ་འཁོབ་པའི་སྐྱེ་བོས་སྤངས་པ། སྐྱེ་བོ་མཁས་པས་བསྟེན་པ། དེ་ན་ཆུ་ཀླུང་གང་གཱ་ཁ་རྗེ་དང་། བསོད་ནམས་དང་། བཀྲ་ཤིས་དང་། གཙང་བ་དང་། གཙང་བ་
ཉིད་ཀྱི་གནས་སུ་གྲགས་པ། འགྲམ་གཉིས་ནས་རླན་འཕྱོ་བ། དྲང་སྲོང་རྣམས་ཀྱི་བསྟི་གནས་ཞེས་བྱ་བ་ཆུ་དགུར་བཅྭ་བརྒྱད་འབབ་པ་ཡོད་དེ། དེར་དྲང་སྲོང་རྣམས་ཀྱི་དཀའ་ཐུབ་སྤྱད་ན་རང་གི་ལུས་ཉིད་ཀྱིས་མཐོ་རིས་སུ་འདོང་བ་ཡིན་ནོ། །ཀྱེ་བྲམ་ཟེའི་ཁྱེའུ་དེ་ནི་
