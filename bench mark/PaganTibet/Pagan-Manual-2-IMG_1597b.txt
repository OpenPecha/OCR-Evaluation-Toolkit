༄༅། ༸ །བསལ། སྤོ་ཆུ་སྒྲ་བླ་གཉན་པོ་རྣཾས་། གསངས་གིས་མོགས་འདྲེ་སིལ་
གིས་གསལ། བྱ་མ་བྱིས་དང་རྒྱ་འང་དང་། རྟོེའུ་ལོ་ཉེས་ལ། ས་བ་མཚན་བཟང་ཀླུ་ཏུ་
བགྲངས་། ར་ཡུ་ལུག་ཡུ་འགུར་དུ་མོག་འདྲེ་བརྨག་བྱས་བཏག་། མོགས་ཡས་མང་པོ་
དྱབ་དུ་བཀལ། ལྷ་མཚན་སྒྲ་བླའི་ཡུལ་འཁཾས་ལེས། མོག་འདྲེ་བ༹ཀ༹་བསལ་གིས་
བཀའ་སྒྱུར་བྱས་། དམུ་དག་ལྷ་དག་ཀོར་ལ། རྨང་གིས་མདུད་པ་དགུ་འཇལ་དེས། ཕོ་
རྨུང་དགུའི་ཀྲོལ་བཞིན་དུ་། འདྲེ་དང་བྱུར་སྲིན་ཁོ་ཐག་བཅད་། སྤྱི་བོ་རྟོ་ལེབ་བསྲ་བ་བཀོད་།
